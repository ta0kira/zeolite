struct Type_Value : public TypeInstance {
  std::string CategoryName() const { return "Value"; }
  Category_Value& parent;
  bool CanConvertFrom(const TypeInstance& from) const final {
    std::vector<const TypeInstance*> args;
    if (!from.TypeArgsForParent(parent, args)) return false;
    if(args.size() != 0) {
      FAIL() << "Wrong number of args (" << args.size() << ")  for " << CategoryName();
    }
    return true;
  }
  bool TypeArgsForParent(const TypeCategory& category, std::vector<const TypeInstance*>& args) const final {
    if (&category == &GetCategory_Value()) {
      args = std::vector<const TypeInstance*>{};
      return true;
    }
    if (&category == &GetCategory_Formatted()) {
      args = std::vector<const TypeInstance*>{};
      return true;
    }
    return false;
  }
  Type_Value(Category_Value& p, Params<0>::Type params) : parent(p) {
    CycleCheck<Type_Value>::Check();
    CycleCheck<Type_Value> marker(*this);
    TRACE_FUNCTION("Value (init @type)")
  }
  ReturnTuple Dispatch(const DFunction<SymbolScope::TYPE>& label, const ParamTuple& params, const ValueTuple& args) final {
    using CallType = ReturnTuple(Type_Value::*)(const ParamTuple&, const ValueTuple&);
    static const CallType Table_Value[] = {
      &Type_Value::Call_new,
    };
    if (label.Collection() == Functions_Value) {
      if (label.Num() < 0 || label.Num() >= 1) {
        FAIL() << "Bad function call " << label.FunctionName();
      }
      return (this->*Table_Value[label.Num()])(params, args);
    }
    return TypeInstance::Dispatch(label, params, args);
  }
  ReturnTuple Call_new(const ParamTuple& params, const ValueTuple& args) {
    TRACE_FUNCTION("Value.new")
    ReturnTuple returns(1);
    SET_CONTEXT_POINT("\"example/tree-demo.0rx\" (line 117, column 12)")
    return ReturnTuple(CreateValue(*this, ParamTuple(), ArgTuple()));
    return returns;
  }
};
Type_Value& CreateType(Params<0>::Type params) {
  static auto& cache = *new InstanceMap<0,Type_Value>();
  auto& cached = cache[params];
  if (!cached) { cached = R_get(new Type_Value(CreateCategory(), params)); }
  return *cached;
}
