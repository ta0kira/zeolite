interface Read<|x> {
  x requires Read<x>
}

interface Write<y|> {
  y requires Write<y>
}

// Extending inheritance a bit.
interface Write2<k|> {
  refines Write<k>
  k requires Write<k>
}

interface Queue<z> {
  refines Write<z>
  refines Read<z>
  z requires Write<z>
  z requires Read<z>
}

interface Queue2<q> {
  refines Write<q>
  refines Read<q>
  // Meets Write<q> requirement, but not directly.
  q requires Write2<q>
  q requires Read<q>
}


concrete Value {
  refines Read<Value>
  refines Write2<Value>
}

concrete Value2 {}


// Just a context for parsing instances.

  x requires Write<x>
  x requires Read<x>
}
