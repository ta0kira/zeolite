interface Read<|x> {
  x requires Read<x>
}

interface Write<x|> {
  x requires Write2<x>
}

interface Write2<x|> {
  refines Write<x>
  x requires Write<x>
}

interface Queue<x> {
  refines Write<x>
  refines Read<x>
  // Requires Write2<x>, not Write<x>.
  x requires Write<x>
  x requires Read<x>
}
