struct Type_KVWriter : public TypeInstance {
  std::string CategoryName() const final { return "KVWriter"; }
  std::string TypeName() const final {
    return TypeInstance::TypeNameFrom("KVWriter",Param_k,Param_v);
  }
  Category_KVWriter& parent;
  TypeInstance& Param_k;
  TypeInstance& Param_v;
  bool CanConvertFrom(const TypeInstance& from) const final {
    std::vector<const TypeInstance*> args;
    if (!from.TypeArgsForParent(parent, args)) return false;
    if(args.size() != 2) {
      FAIL() << "Wrong number of args (" << args.size() << ")  for " << CategoryName();
    }
    if (!TypeInstance::CanConvert(Param_k, *args[0])) return false;
    if (!TypeInstance::CanConvert(Param_v, *args[1])) return false;
    return true;
  }
  bool TypeArgsForParent(const TypeCategory& category, std::vector<const TypeInstance*>& args) const final {
    if (&category == &GetCategory_KVWriter()) {
      args = std::vector<const TypeInstance*>{&Param_k,&Param_v};
      return true;
    }
    return false;
  }
  Type_KVWriter(Category_KVWriter& p, Params<2>::Type params) : parent(p), Param_k(*std::get<0>(params)), Param_v(*std::get<1>(params)) {}
};
Type_KVWriter& CreateType(Params<2>::Type params) {
  static auto& cache = *new InstanceMap<2,Type_KVWriter>();
  auto& cached = cache[params];
  if (!cached) { cached = R_get(new Type_KVWriter(CreateCategory(), params)); }
  return *cached;
}
}
TypeCategory& GetCategory_KVWriter() {
