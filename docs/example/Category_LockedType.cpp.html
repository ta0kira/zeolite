struct Category_LockedType : public TypeCategory {
  std::string CategoryName() const final { return "LockedType"; }
  Category_LockedType() {
    CycleCheck<Category_LockedType>::Check();
    CycleCheck<Category_LockedType> marker(*this);
    TRACE_FUNCTION("LockedType (init @category)")
  }
  ReturnTuple Dispatch(const DFunction<SymbolScope::CATEGORY>& label, const ParamTuple& params, const ValueTuple& args) final {
    using CallType = ReturnTuple(Category_LockedType::*)(const ParamTuple&, const ValueTuple&);
    return TypeCategory::Dispatch(label, params, args);
  }
};
Category_LockedType& CreateCategory() {
  static auto& category = *new Category_LockedType();
  return category;
}
struct Type_LockedType : public TypeInstance {
  std::string CategoryName() const final { return parent.CategoryName(); }
  void BuildTypeName(std::ostream& output) const final {
    return TypeInstance::TypeNameFrom(output, parent, Param_x);
  }
  Category_LockedType& parent;
  TypeInstance& Param_x;
  bool CanConvertFrom(const TypeInstance& from) const final {
    std::vector<const TypeInstance*> args;
    if (!from.TypeArgsForParent(parent, args)) return false;
    if(args.size() != 1) {
      FAIL() << "Wrong number of args (" << args.size() << ")  for " << CategoryName();
    }
    if (!TypeInstance::CanConvert(*args[0], Param_x)) return false;
    return true;
  }
  bool TypeArgsForParent(const TypeCategory& category, std::vector<const TypeInstance*>& args) const final {
    if (&category == &GetCategory_LockedType()) {
      args = std::vector<const TypeInstance*>{&Param_x};
      return true;
    }
    return false;
  }
  Type_LockedType(Category_LockedType& p, Params<1>::Type params) : parent(p), Param_x(*std::get<0>(params)) {
    CycleCheck<Type_LockedType>::Check();
    CycleCheck<Type_LockedType> marker(*this);
    TRACE_FUNCTION("LockedType (init @type)")
  }
  ReturnTuple Dispatch(const DFunction<SymbolScope::TYPE>& label, const ParamTuple& params, const ValueTuple& args) final {
    using CallType = ReturnTuple(Type_LockedType::*)(const ParamTuple&, const ValueTuple&);
    static const CallType Table_LockedType[] = {
      &Type_LockedType::Call_create,
    };
    if (label.Collection() == Functions_LockedType) {
      if (label.Num() < 0 || label.Num() >= 1) {
        FAIL() << "Bad function call " << label.FunctionName();
      }
      return (this->*Table_LockedType[label.Num()])(params, args);
    }
    return TypeInstance::Dispatch(label, params, args);
  }
  ReturnTuple Call_create(const ParamTuple& params, const ValueTuple& args) {
    TRACE_FUNCTION("LockedType.create")
    ReturnTuple returns(1);
    const S<TypeValue>& Var_k = (args.At(0));
    const S<TypeValue>& Var_v = (args.At(1));
    SET_CONTEXT_POINT("\"example/type-tree.0rx\" (line 75, column 5)")
    return ReturnTuple(CreateValue(*this, ParamTuple(), ArgTuple((Var_k), (Var_v))));
    return returns;
  }
};
Type_LockedType& CreateType(Params<1>::Type params) {
  static auto& cache = *new InstanceMap<1,Type_LockedType>();
  auto& cached = cache[params];
  if (!cached) { cached = R_get(new Type_LockedType(CreateCategory(), params)); }
  return *cached;
}
struct Value_LockedType : public TypeValue {
  Value_LockedType(Type_LockedType& p, const ParamTuple& params, const ValueTuple& args) : parent(p), Var_key((args.At(0))), Var_value((args.At(1))) {}
  ReturnTuple Dispatch(const S<TypeValue>& self, const DFunction<SymbolScope::VALUE>& label, const ParamTuple& params,const ValueTuple& args) final {
    using CallType = ReturnTuple(Value_LockedType::*)(const S<TypeValue>&, const ParamTuple&, const ValueTuple&);
    static const CallType Table_LockedType[] = {
      &Value_LockedType::Call_check,
    };
    if (label.Collection() == Functions_LockedType) {
      if (label.Num() < 0 || label.Num() >= 1) {
        FAIL() << "Bad function call " << label.FunctionName();
      }
      return (this->*Table_LockedType[label.Num()])(self, params, args);
    }
    return TypeValue::Dispatch(self, label, params, args);
  }
  std::string CategoryName() const final { return parent.CategoryName(); }
  ReturnTuple Call_check(const S<TypeValue>& Var_self, const ParamTuple& params, const ValueTuple& args) {
    TRACE_FUNCTION("LockedType.check")
    ReturnTuple returns(1);
    TypeInstance& Param_y = *params.At(0);
    const S<TypeValue>& Var_k = (args.At(0));
    if (PRED_CONTEXT_POINT("\"example/type-tree.0rx\" (line 79, column 5)")(GetType_TypeKey(T_get(&GetMerged_Any())).Call(Function_Equals_equals, ParamTuple(), ArgTuple((Var_key), (Var_k)))).Only()->AsBool()) {
    SET_CONTEXT_POINT("\"example/type-tree.0rx\" (line 88, column 7)")
    return ReturnTuple(TypeInstance::Reduce(parent.Param_x, Param_y, (Var_value)));
    }
    else {
    SET_CONTEXT_POINT("\"example/type-tree.0rx\" (line 90, column 7)")
    return ReturnTuple(Var_empty);
    }
    return returns;
  }
  S<TypeValue> Var_key;
  S<TypeValue> Var_value;
  Type_LockedType& parent;
};
S<TypeValue> CreateValue(Type_LockedType& parent, const ParamTuple& params, const ValueTuple& args) {
  return S_get(new Value_LockedType(parent, params, args));
}
