interface Read<|x> {
  x requires Read<x>
}

interface Write<x|> {
  x requires Write2<x>
}

interface Write2<x|> {
  inherits Write<x>
  x requires Write<x>
}

interface Queue<x> {
  inherits Write<x>
  inherits Read<x>
  // Requires Write2<x>, not Write<x>.
  x requires Write<x>
  x requires Read<x>
}
